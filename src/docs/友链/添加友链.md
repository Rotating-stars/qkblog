<script setup>
import {
	VPTeamPage,
	VPTeamPageTitle,
	VPTeamMembers,
	VPTeamPageSection
} from 'vitepress/theme'
import qk from './头像/qk.jpg'

import { ref } from 'vue'
import { ElMessage } from 'element-plus'
// 表单ref对象
const formRef = ref(null)
// 表单数据
const formDate = ref({
	avatar: '',
	name: '',
	title: '',
	org: '',
	orgLink: '',
	desc: '',
	links: [],
})
// 校验规则
const rules = {
  name: [
    { required: true, message: '请输入昵称', trigger: 'blur' }
  ],
  org: [
    { required: true, message: '请输入友站名称', trigger: 'blur' }
  ],
  orgLink: [
    { required: true, message: '请输入友站链接', trigger: 'blur' }
  ],
  links: [
    { 
      type: 'array', 
      max: 4, 
      message: '社交链接最多4个', 
      trigger: 'change' 
    }
  ]
}

const linksOptions = [
	{name: 'Github', icon: 'github', link: ''},
	{name: 'RSS', icon: 'rss', link: ''},
	{name: '邮箱', icon: 'maildotru', link: ''},
	{name: 'Bilibili', icon: 'bilibili', link: ''},
	{name: '知乎', icon: 'zhihu', link: ''},
	{name: '微博', icon: 'sinaweibo', link: ''},
	{name: '小红书', icon: 'xiaohongshu', link: ''},
	{name: 'X', icon: 'x', link: ''},
	{name: 'YouTube', icon: 'youtube', link: ''},
]

// 重置方法
const resetForm = () => {
	formDate.value = {
		avatar: '',
  		name: '',
  		title: '',
  		org: '',
  		orgLink: '',
  		desc: '',
  		links: [{icon: "", link: ""}],
	};
	// 重置表单校验状态
	formRef.value?.clearValidate();
	ElMessage.primary('已重置！');
}
// 脚本字符串
const createBatString = () => {
	const batString = JSON.stringify(formDate.value);
	return batString
}
// 复制方法
const copyBatString = async () => {
	// 复制前校验
	if (!formRef.value) return
	try {
		await formRef.value.validate()
		// 调用浏览器复制API
		await navigator.clipboard.writeText(createBatString());
		ElMessage.success('复制成功！');
	} catch (err) {
		ElMessage.warning('请正确填写后再复制！');
	}
}

const partners = [
	{
		avatar: qk, name: "昵称", title: "标签",
		org: "友站名称", orgLink: "友站链接", 
		desc: "一句话介绍",
		links: [{icon: 'bilibili', link: '社交链接'},
				{icon: 'github', link: '社交链接'},],
	},
];
</script>

<VPTeamPageSection>
	<template #title>交换友链</template>
	<template #lead>请将以下信息的json（脚本复制）发送邮件给我（右上角邮箱标志）<br>若无头像链接请带附件（不大于256*256的jpg文件）</template>
    <template #members>
      <VPTeamMembers size="small" :members="partners" />
    </template>
</VPTeamPageSection>

<el-card shadow="hover" style="max-width: 800px; margin: 0 auto; margin-top: 20px; margin-bottom: 20px;">
	<template #header>
    	<div class="card-header">
    		<span>友链信息填写器</span>
    	</div>
	</template>
  	<el-form
  	  	ref="formRef"
  	  	:model="formDate"
  	  	:rules="rules"
  	  	label-position="right"
  	  	label-width="130px"
	>
		<el-form-item label="头像链接：" prop="">
  	  	  <el-input v-model="formDate.avatar" placeholder="或者你直接邮箱发图给我"/>
  	  	</el-form-item>
  	  	<el-form-item label="昵称：" prop="name">
  	  	  <el-input v-model="formDate.name" placeholder="示例：青空"/>
  	  	</el-form-item>
		<el-form-item label="标签：" prop="">
  	  	  <el-input v-model="formDate.title" placeholder="示例：本站作者"/>
  	  	</el-form-item>
		<el-form-item label="友站名称：" prop="org">
  	  	  <el-input v-model="formDate.org" placeholder="示例：青空的Blog"/>
  	  	</el-form-item>
		<el-form-item label="友站链接：" prop="orgLink">
  	  	  <el-input v-model="formDate.orgLink" placeholder="示例：https://github.com/Rotating-stars"/>
  	  	</el-form-item>
		<el-form-item label="介绍：" prop="">
  	  	  <el-input v-model="formDate.desc" placeholder="示例：谨以此书纪念我的童年，那是一段小有遗憾的幸福时光。"/>
  	  	</el-form-item>
		<el-form-item label="社交链接：" prop="links">
      		<el-checkbox-group v-model="formDate.links">
        		<el-checkbox v-for="i in linksOptions" :value="i" name="type">{{i.name}}</el-checkbox>
      		</el-checkbox-group>
    	</el-form-item>
		<el-form-item v-for="(i, index) in formDate.links"
			:label="i.icon+'：'" :prop="'links.'+index+'.link'" :rules="[{ required: true, message: '请输入链接', trigger: 'blur' }]">
  	  		<el-input v-model="i.link" :placeholder="i.name+'链接'"/>
  	  	</el-form-item>
  	</el-form>
  	<template #footer>
		<el-button type="primary" @click="copyBatString()">点击复制</el-button>
		<el-button @click="resetForm()">重置</el-button>
  	</template>
</el-card>
